CREATE TABLE cliente (
Idcliente INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
Idcontoclient INTEGER,
Userclient VARCHAR(128),
Clientpsw VARCHAR(128);

CREATE TABLE venditore (
Idseller INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
Userseller VARCHAR(128),
Sellerpsw VARCHAR(128);

CREATE TABLE oggetto (
objid INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
objname VARCHAR(128),
objurl VARCHAR(128),
objdesc VARCHAR(128),
objprice FLOAT,
objdisp INTEGER,
objsell INTEGER,
FOREIGN KEY (IdSELLER) REFERENCES venditore(Idseller);

CREATE TABLE saldo (
idsaldo INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
saldo FLOAT,
FOREIGN KEY (Idcliente) REFERENCES cliente(Idcliente);
